const cleanData = (data: any) => {
  const cleaned: any = {}
  Object.keys(data).forEach(key => {
    if (data[key] === undefined || data[key] === '') {
      cleaned[key] = null
    } else {
      cleaned[key] = data[key]
    }
  })
  return cleaned
}

  const addMember = async (memberData: MemberForm) => {
